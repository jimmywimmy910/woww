local toggleOn = false
local autoCollectOn = false

local function teleportToLootBoxes()
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end
    local LootsFolder = Workspace:FindFirstChild("Loots")
    if not LootsFolder then return end
    for _, lootBox in ipairs(LootsFolder:GetChildren()) do
        if lootBox.Name == "CashLootBox" and lootBox:IsA("Model") and lootBox.PrimaryPart then
            lootBox:SetPrimaryPartCFrame(LocalPlayer.Character.HumanoidRootPart.CFrame)
        end
    end
end

local function touchCashCollector()
    if not Workspace:FindFirstChild("CashZone") then return end
    local CashZone = Workspace.CashZone
    for _, collector in ipairs(CashZone:GetDescendants()) do
        if collector.Name == "CashCollector" and collector:IsA("BasePart") then
            firetouchinterest(LocalPlayer.Character.HumanoidRootPart, collector, 0)
            firetouchinterest(LocalPlayer.Character.HumanoidRootPart, collector, 1)
        end
    end
end

local function toggleSwitch()
    toggleOn = not toggleOn
    if toggleOn then
        ToggleSwitchButton.Position = UDim2.new(0.5, 2, 0, 2)
        ToggleSwitchButton.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        while toggleOn do
            teleportToLootBoxes()
            wait(1) -- Adjust the time interval as needed
        end
    else
        ToggleSwitchButton.Position = UDim2.new(0, 2, 0, 2)
        ToggleSwitchButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    end
end

local function toggleAutoCollect()
    autoCollectOn = not autoCollectOn
    if autoCollectOn then
        AutoCollectSwitchButton.Position = UDim2.new(0.5, 2, 0, 2)
        AutoCollectSwitchButton.BackgroundColor3 = Color3.fromRGB(0, 255, 0)
        while autoCollectOn do
            touchCashCollector()
            wait(1) -- Adjust the time interval as needed
        end
    else
        AutoCollectSwitchButton.Position = UDim2.new(0, 2, 0, 2)
        AutoCollectSwitchButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    end
end

ToggleSwitchButton.MouseButton1Click:Connect(toggleSwitch)
AutoCollectSwitchButton.MouseButton1Click:Connect(toggleAutoCollect)

ExitButton.MouseButton1Click:Connect(function()
    ScreenGui:Destroy()
end)

-- Enable dragging of the main UI
local dragging = false
local dragStart
local startPos

local function updateInput(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position

        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        UserInputService.InputChanged:Connect(updateInput)
    end
end)
