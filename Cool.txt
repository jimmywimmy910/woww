-- Function to show notification
local function showNotification()
    NotificationLabel.Text = "Join Our Roblox Group for more scripts!"
    NotificationFrame.Visible = true
    local tweenInfo = TweenInfo.new(1, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut)
    local goalIn = {Position = UDim2.new(0.5, -150, 1, -150)}
    local tweenIn = TweenService:Create(NotificationFrame, tweenInfo, goalIn)
    tweenIn:Play()

    wait(5) -- Show for 5 seconds

    local goalOut = {Position = UDim2.new(1, 100, 1, -100)}
    local tweenOut = TweenService:Create(NotificationFrame, tweenInfo, goalOut)
    tweenOut:Play()

    tweenOut.Completed:Connect(function()
        NotificationFrame.Visible = false
    end)
end

-- Function to copy link to clipboard
local function copyToClipboard(text)
    local clipboard = Instance.new("StringValue")
    clipboard.Name = "Clipboard"
    clipboard.Value = text
    clipboard.Parent = PlayerGui
    clipboard:Destroy()
end

NotificationButton.MouseButton1Click:Connect(function()
    copyToClipboard(groupLink)
end)

-- Play notification sound every 10 minutes
local function playNotificationSound()
    local sound = Instance.new("Sound")
    sound.SoundId = notificationSoundId
    sound.Parent = SoundService
    sound:Play()
end

-- Check if player is in group
local function checkGroupMembership()
    local success, isInGroup = pcall(function()
        return LocalPlayer.MembershipType == Enum.MembershipType.Premium or LocalPlayer:HasGroup(groupId)
    end)

    if not success or not isInGroup then
        LocalPlayer:Kick("Join the group to use this script: " .. groupLink)
    end
end

-- Show notification every 10 minutes
while true do
    playNotificationSound()
    showNotification()
    wait(600) -- 10 minutes
end

-- Show credits frame on button click
CreditsButton.MouseButton1Click:Connect(function()
    CreditsFrame.Visible = not CreditsFrame.Visible
end)

-- Check group membership
checkGroupMembership()
